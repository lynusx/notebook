# 操作符

操作符是编程语言中用于执行特定操作的符号。Dart 提供了丰富的操作符，包括算术、关系、逻辑、位运算、类型测试等多种类型。

## 第1章 算术操作符

### 1.1 基本算术

```dart
void main() {
  int a = 10;
  int b = 3;
  
  // 加法
  print(a + b);  // 13
  
  // 减法
  print(a - b);  // 7
  
  // 乘法
  print(a * b);  // 30
  
  // 除法（返回 double）
  print(a / b);  // 3.3333333333333335
  
  // 整除（返回 int）
  print(a ~/ b);  // 3
  
  // 取模（余数）
  print(a % b);  // 1
}
```

### 1.2 自增自减

```dart
// 示例 1
void main() {
  int count = 0;
  
  // 前置++：先自增，返回新值
  print(++count);  // 1
  print(count);    // 1
  
  // 后置++：先返回原值，后自增
  print(count++);  // 1
  print(count);    // 2
  
  // 前置--：先自减，返回新值
  print(--count);  // 1
  
  // 后置--：先返回原值，后自减
  print(count--);  // 1
  print(count);    // 0
}

// 示例 2
void prefixVsPostfix() {
  var a = 0;
  var b = 0;
  
  var x = ++a;  // a = 1, x = 1
  var y = b++;  // y = 0, b = 1
  
  print('a=$a, x=$x');  // a=1, x=1
  print('b=$b, y=$y');  // b=1, y=0
}
```

### 1.3 取反

```dart
void main() {
  int a = 5;
  
  // 一元取反
  print(-a);  // -5
}
```

### 1.4 字符串拼接

```dart
void main() {
  String firstName = 'Alice';
  String lastName = 'Smith';
  
  // + 操作符拼接字符串
  String fullName = firstName + ' ' + lastName;
  print(fullName);  // Alice Smith
  
  // 字符串插值（推荐）
  String greeting = 'Hello, $firstName $lastName!';
  print(greeting);  // Hello, Alice Smith!
  
  // 表达式插值
  int age = 25;
  String info = 'Next year, I will be ${age + 1}';
  print(info);  // Next year, I will be 26
}
```

## 第2章 关系与逻辑操作符

### 2.1 相等关系操作符

```dart
void main() {
  int a = 10;
  int b = 20;
  
  // 等于
  print(a == b);  // false
  
  // 不等于
  print(a != b);  // true
  
  // 大于
  print(a > b);   // false
  
  // 小于
  print(a < b);   // true
  
  // 大于等于
  print(a >= b);  // false
  
  // 小于等于
  print(a <= b);  // true
}
```

### 2.2 逻辑运算

```dart
void main() {
  bool isLoggedIn = true;
  bool isAdmin = false;
  
  // 逻辑与（AND）
  print(isLoggedIn && isAdmin);  // false
  
  // 逻辑或（OR）
  print(isLoggedIn || isAdmin);  // true
  
  // 逻辑非（NOT）
  print(!isLoggedIn);  // false
  
  // 短路求值
  bool result = isLoggedIn && expensiveOperation();
  // 如果 isLoggedIn 为 false，expensiveOperation 不会执行
}

bool expensiveOperation() {
  print('Expensive operation executed');
  return true;
}
```

### 2.3 类型测试

```dart
void main() {
  dynamic employee = Person('Default');

  // is：检查类型
  if (employee is Person) {
    employee.firstName = 'Alice';
  }

  // is!：检查不是某类型
  if (employee is! int) {
    print('employee is not int');
  }

  // as：类型转换
  (employee as Person).firstName = 'Bob';
}

class Person {
  String firstName;
  Person(this.firstName);
}
```

## 第3章 位运算与移位操作符

### 3.1 位运算

```dart
void main() {
  int a = 5;   // 二进制: 0101
  int b = 3;   // 二进制: 0011
  
  // 按位与（AND）
  print(a & b);  // 1  (0001)
  
  // 按位或（OR）
  print(a | b);  // 7  (0111)
  
  // 按位异或（XOR）
  print(a ^ b);  // 6  (0110)
  
  // 按位取反（NOT）
  print(~a);     // -6 (补码表示)
}
```

### 3.2 移位

```dart
void main() {
  int a = 8;   // 二进制: 1000
  
  // 左移（<<）：乘以 2^n
  print(a << 1);   // 16  (10000)
  print(a << 2);   // 32  (100000)
  
  // 右移（>>）：除以 2^n
  print(a >> 1);   // 4   (0100)
  print(a >> 2);   // 2   (0010)
  
  // 无符号右移（>>>）：Dart 2.14+
  int negative = -8;
  print(negative >>> 1);  // 大正数（高位补0）
}
```

### 3.3 位运算应用场景

```dart
// 权限系统
class Permissions {
  static const int read = 1;      // 0001
  static const int write = 2;     // 0010
  static const int execute = 4;   // 0100
  static const int delete = 8;    // 1000
}

void permissionExample() {
  int userPermissions = Permissions.read | Permissions.write;
  
  // 检查权限
  bool canRead = userPermissions & Permissions.read != 0;
  bool canDelete = userPermissions & Permissions.delete != 0;
  
  print('Can read: $canRead');      // true
  print('Can delete: $canDelete');  // false
  
  // 添加权限
  userPermissions |= Permissions.execute;
  
  // 移除权限
  userPermissions &= ~Permissions.write;
  
  // 切换权限
  userPermissions ^= Permissions.delete;
}

// 标志位操作
class FileFlags {
  static const int readable = 1 << 0;
  static const int writable = 1 << 1;
  static const int executable = 1 << 2;
  static const int hidden = 1 << 3;
}
```

## 第4章 赋值与条件操作符

### 4.1 赋值

```dart
void main() {
  int a = 10;

  // 简单赋值
  num b = a;

  // 复合赋值
  b += 5; // b = b + 5
  b -= 3; // b = b - 3
  b *= 2; // b = b * 2
  b /= 2; // b = b / 2;
  b ~/= 2; // b = b ~/ 2
  b %= 3; // b = b % 3

  // 位运算复合赋值
  int flags = 0;
  flags |= 1; // 设置位
  flags &= ~1; // 清除位
  flags ^= 1; // 切换位
}
```

### 4.2 空合并赋值

```dart
void main() {
  String? name;
  
  // ??=：仅在变量为 null 时赋值
  name ??= 'Default';
  print(name);  // Default
  
  name ??= 'Other';  // 不会执行，因为 name 不为 null
  print(name);  // Default
  
  // 对比普通赋值
  String? value;
  value = 'First';
  value = 'Second';  // 会覆盖
  print(value);  // Second
}
```

### 4.3 条件表达式

```dart
void main() {
  int score = 85;
  
  // 三元操作符
  String result = score >= 60 ? 'Pass' : 'Fail';
  print(result);  // Pass
  
  // 嵌套三元（不推荐，可读性差）
  String grade = score >= 90 ? 'A' 
               : score >= 80 ? 'B'
               : score >= 60 ? 'C'
               : 'F';
  
  // ??：空合并操作符
  String? name;
  String displayName = name ?? 'Unknown';
  print(displayName);  // Unknown
  
  // 链式空合并
  String? firstChoice;
  String? secondChoice;
  String finalChoice = firstChoice ?? secondChoice ?? 'Default';
}
```

### 4.4 展开操作符（...）

```dart
void main() {
  // 列表展开
  var list1 = [1, 2, 3];
  var list2 = [0, ...list1, 4, 5];
  print(list2);  // [0, 1, 2, 3, 4, 5]
  
  // 空感知展开（...?）
  List<int>? nullableList;
  var list3 = [0, ...?nullableList, 4];
  print(list3);  // [0, 4]
  
  // 映射展开
  var map1 = {'a': 1, 'b': 2};
  var map2 = {'c': 3, ...map1};
  print(map2);  // {c: 3, a: 1, b: 2}
  
  // 集合展开
  var set1 = {1, 2, 3};
  var set2 = {0, ...set1, 4};
  print(set2);  // {0, 1, 2, 3, 4}
}
```

## 第5章 级联与访问操作符

### 5.1 级联操作符（..）

```dart
class Person {
  String name = '';
  int age = 0;
  String email = '';
  
  void sayHello() {
    print('Hello, I am $name');
  }
}

void main() {
  // 不使用级联
  var person1 = Person();
  person1.name = 'Alice';
  person1.age = 25;
  person1.email = 'alice@example.com';
  person1.sayHello();
  
  // 使用级联（更简洁）
  var person2 = Person()
    ..name = 'Bob'
    ..age = 30
    ..email = 'bob@example.com'
    ..sayHello();
  
  // 级联返回原对象
  var person3 = Person()
    ..name = 'Charlie'
    ..age = 35;
  
  print(person3.name);  // Charlie
}
```

### 5.2 空感知级联（?..）

```dart
void main() {
  Person? nullablePerson;
  
  // 安全级联调用
  nullablePerson
    ?..name = 'Alice'
    ?..age = 25
    ?..sayHello();  // 不会执行，因为对象为 null
  
  // 非空对象
  Person person = Person();
  person
    ?..name = 'Bob'  // 安全级联也适用
    ..age = 30;
}
```

级联操作符应用场景

```dart
// 配置对象
final paint = Paint()
  ..color = Colors.red
  ..strokeWidth = 2.0
  ..style = PaintingStyle.stroke;

// 链式配置
final textStyle = TextStyle()
  ..fontSize = 16
  ..fontWeight = FontWeight.bold
  ..color = Colors.blue;

// 构建器模式
final query = QueryBuilder()
  ..select(['id', 'name'])

  ..orderBy('name');
```

### 5.3 成员访问操作符

```dart
void main() {
  var person = Person()..name = 'Alice';
  
  // . 成员访问
  print(person.name);
  
  // ?. 空感知成员访问
  Person? nullablePerson;
  print(nullablePerson?.name);  // null，不会抛出异常
  
  // ! 非空断言
  Person? maybePerson = Person()..name = 'Bob';
  print(maybePerson!.name);  // 断言非空
  
  // [] 成员索引
  var list = [1, 2, 3];
  print(list[0]);  // 1
  
  // ?[] 空感知成员索引
  List<int>? nullableList;
  print(nullableList?[0]);  // null
}
```

## 第6章 操作符重载

Dart 允许在类中重载操作符，支持的操作符：`<` `>` `>=` `<=` `=` `+` `-` `*` `/` `~/` `%` `|` `^` `&` `<<` `>>` `>>>` `[]` `[]=` `~` `unary~` 。

```dart
class Vector {
  final int x, y;

  Vector(this.x, this.y);

  // 重载 + 操作符
  Vector operator +(Vector other) => Vector(x + other.x, y + other.y);

  // 重载 == 操作符
  @override
  bool operator ==(Object other) =>
      other is Vector && x == other.x && y == other.y;

  @override
  int get hashCode => Object.hash(x, y);

  @override
  String toString() => 'Vector($x, $y)';
}

void main() {
  final v1 = Vector(2, 3);
  final v2 = Vector(2, 2);
  final v3 = v1 + v2; // Vector(4, 5)
}
```

## 第7章 最佳实践

### 7.1 操作符优先级

```dart
void precedenceExample() {
  // 优先级从高到低：
  // 1. 一元操作符：! ~ ++ -- - ...
  // 2. 乘除：* / ~/ %
  // 3. 加减：+ -
  // 4. 移位：<< >>
  // 5. 位与：&
  // 6. 位异或：^
  // 7. 位或：|
  // 8. 关系：>= > <= <
  // 9. 相等：== !=
  // 10. 逻辑与：&&
  // 11. 逻辑或：||
  // 12. 空合并：??
  // 13. 条件：? :
  // 14. 赋值：= += -= ...
  
  // 使用括号明确优先级
  var result = (a + b) * c;  // 清晰
  var result2 = a + b * c;   // 可能产生歧义
}
```

### 7.2 操作符优先级表

| 优先级 | 操作符 | 描述 |
|--------|--------|------|
| 1 | `()` `[]` `.` `?.` | 括号、索引、成员访问 |
| 2 | `++` `--` `-` `!` `~` | 自增自减、一元操作 |
| 3 | `*` `/` `~/` `%` | 乘除、整除、取模 |
| 4 | `+` `-` | 加减、字符串拼接 |
| 5 | `<<` `>>` `>>>` | 移位 |
| 6 | `&` | 位与 |
| 7 | `^` | 位异或 |
| 8 | `\|` | 位或 |
| 9 | `<=` `<` `>=` `>` | 关系 |
| 10 | `==` `!=` | 相等 |
| 11 | `&&` | 逻辑与 |
| 12 | `\|\|` | 逻辑或 |
| 13 | `??` | 空合并 |
| 14 | `?:` | 条件 |
| 15 | `=` `+=` `-=` ... | 赋值 |

### 7.3 常见陷阱

```dart
void commonPitfalls() {
  // 陷阱 1：整数除法
  int a = 5;
  int b = 2;
  print(a / b);   // 2.5 (double)
  print(a ~/ b);  // 2 (int)
  
  // 陷阱 2：空感知链
  String? name;
  // print(name?.length + 1);  // 错误：int? 不能 + int
  print((name?.length ?? 0) + 1);  // ✅ 正确
  
  // 陷阱 3：赋值操作符返回值
  int x = 5;
  int y = 10;
  if ((x = y) > 5) {  // 赋值后比较
    print('x = $x');  // x = 10
  }
  
  // 陷阱 4：位运算优先级
  int flags = 1 | 2 & 3;  // 等价于 1 | (2 & 3) = 3
  
  // 陷阱 5：短路求值的副作用
  bool condition = false;
  condition && expensiveFunction();  // expensiveFunction 不会执行
}

bool expensiveFunction() {
  print('Expensive!');
  return true;
}
```

### 7.4 最佳实践总结

```dart
// ✅ 使用括号提高可读性
bool isValid = (age >= 18) && (age <= 65);

// ✅ 使用空合并提供默认值
String displayName = userName ?? 'Anonymous';

// ✅ 使用级联简化连续操作
final config = Config()
  ..host = 'localhost'
  ..port = 8080;

// ✅ 使用展开合并集合
final combined = [...list1, ...list2];

// ✅ 使用三元操作符简化简单条件
String status = isActive ? 'Active' : 'Inactive';

// ❌ 避免嵌套过深的条件
// 不好：result = a ? b ? c ? d : e : f : g;
// 好：使用 if-else 或提前返回
```
